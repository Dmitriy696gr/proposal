    <script>
        // === СТАРТОВЫЙ ЭКРАН И МУЗЫКА ===
        const startScreen = document.getElementById('start-screen');
        const startBtn = document.getElementById('start-btn');
        const bgMusic = document.getElementById('bg-music');

        startBtn.addEventListener('click', function() {
            // Запускаем музыку
            bgMusic.volume = 0.5; // Громкость 50% (можно изменить от 0 до 1)
            bgMusic.play().catch(e => console.log('Autoplay blocked:', e));
            
            // Скрываем стартовый экран
            startScreen.classList.add('hidden');
            
            // Прокручиваем к первому слайду
            setTimeout(() => {
                document.querySelector('.slide').scrollIntoView({ behavior: 'smooth' });
            }, 500);
        });

        // === АНИМАЦИЯ ПОЯВЛЕНИЯ ТЕКСТА ПРИ СКРОЛЛЕ ===
        const observerOptions = { threshold: 0.3 };
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('visible');
                }
            });
        }, observerOptions);

        document.querySelectorAll('.text-content').forEach(el => observer.observe(el));

        // === ЛОГИКА ВИДЕО И ВОПРОСА ===
        const video = document.getElementById('myVideo');
        const questionBlock = document.getElementById('question-block');

        video.addEventListener('ended', function() {
            questionBlock.classList.add('visible');
            setTimeout(() => {
                questionBlock.scrollIntoView({ behavior: 'smooth', block: 'center' });
            }, 300);
        });

        // === ЛОГИКА КНОПОК ===
        const noBtn = document.getElementById('no-btn');
        
        noBtn.addEventListener('mouseover', moveButton);
        noBtn.addEventListener('touchstart', function(e) {
            e.preventDefault(); 
            moveButton();
        });

        function moveButton() {
            const maxX = window.innerWidth - noBtn.offsetWidth - 20;
            const maxY = window.innerHeight - noBtn.offsetHeight - 20;
            const randomX = Math.max(0, Math.random() * maxX);
            const randomY = Math.max(0, Math.random() * maxY);
            
            noBtn.style.position = 'fixed';
            noBtn.style.left = randomX + 'px';
            noBtn.style.top = randomY + 'px';
            noBtn.style.zIndex = '1000';
        }

        // === ОБРАБОТЧИК КНОПКИ "ДА" ===
        document.getElementById('yes-btn').addEventListener('click', function() {
            questionBlock.style.display = 'none';
            
            const finalMessage = document.getElementById('final-message');
            finalMessage.style.display = 'block';
            
            // Увеличиваем громкость музыки
            bgMusic.volume = 0.8;
            
            // Конфетти
            const duration = 5000;
            const animationEnd = Date.now() + duration;
            const defaults = { startVelocity: 30, spread: 360, ticks: 60, zIndex: 2000 };

            function randomInRange(min, max) {
                return Math.random() * (max - min) + min;
            }

            const interval = setInterval(function() {
                const timeLeft = animationEnd - Date.now();

                if (timeLeft <= 0) {
                    return clearInterval(interval);
                }

                const particleCount = 50 * (timeLeft / duration);
                
                confetti(Object.assign({}, defaults, {
                    particleCount,
                    origin: { x: randomInRange(0.1, 0.3), y: Math.random() - 0.2 }
                }));
                confetti(Object.assign({}, defaults, {
                    particleCount,
                    origin: { x: randomInRange(0.7, 0.9), y: Math.random() - 0.2 }
                }));
            }, 250);

            setTimeout(() => {
                confetti({
                    particleCount: 200,
                    spread: 160,
                    origin: { y: 0.6 },
                    zIndex: 2000
                });
            }, 500);
        });
    </script>
